💻 Configuración con Detección Automática

window.GUIDERS_CONFIG = {
  apiKey: 'your-api-key',
  activeHours: {
    enabled: true,
    timezone: 'auto',  // 🔥 Detección automática!
    ranges: [
      { start: '08:00', end: '14:00' },
      { start: '15:00', end: '17:00' }
    ],
    fallbackMessage: 'Chat disponible de 8:00-14:00 y 15:00-17:00 en tu zona horaria local.'
  }
};
